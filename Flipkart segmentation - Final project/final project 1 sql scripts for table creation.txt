with cte1 as (
select order_id, sum(final_price)as K_1 from order_items group by order_id)
select C.customer_id, current_date - max (O.order_date) as recency, count (O.order_id) as frequency, sum (C1.K_1) from 
customers C join orders O on C.customer_id = O.customer_id join cte1 C1 on O.order_id = C1.order_id
group by C.customer_id





select 
    c.customer_id,
    c.full_name,      
    o.order_id,
    o.order_date,
    oi.product_id,
    oi.quantity,
    oi.final_price
from customers c
join orders o on c.customer_id = o.customer_id
join order_items oi on o.order_id = oi.order_id;
